export const resolveColor = (val) => {
  if (!val) return 'transparent';
  if (val.startsWith('#')) return val;
  if (val.startsWith('{')) {
    const path = val.replace(/[{}]/g, '').split('.');
    let current = tokens.global;
    for (const part of path) {
      if (current && current[part]) {
        current = current[part];
      } else {
        return 'transparent';
      }
    }
    return typeof current === 'string' ? current : current?.$value || 'transparent';
  }
  return val;
};

export const getContrastColor = (hex) => {
  if (!hex || hex === 'transparent' || !hex.startsWith('#')) return 'black';
  try {
    const r = parseInt(hex.slice(1, 3), 16);
    const g = parseInt(hex.slice(3, 5), 16);
    const b = parseInt(hex.slice(5, 7), 16);
    const yiq = (r * 299 + g * 587 + b * 114) / 1000;
    return yiq >= 128 ? 'black' : 'white';
  } catch (e) {
    return 'black';
  }
};

## 색상 팔레트

색상 팔레트는 웹사이트에서 사용되는 색상의 집합을 의미합니다. 색상 팔레트 안에서 시스템 컬러, 브랜드 컬러, 텍스트 컬러, 배경 컬러, 테마 컬러 등 다양한 컬러를 사용하여, 색상의 통일성을 유지합니다.
