# 토스트 알림

사용자에게 시스템의 상태나 작업의 결과를 간략하게 알려주는 피드백 요소입니다. 화면 우측 하단 등 지정된 위치에 일시적으로 나타났다가 사라집니다.

---

<PreviewLayout
  preview={<ToastDemo />}
  code={`import { useToast } from "@/app/templates/Toast/Toast";
import { Button } from "@/app/templates/Button/Button";

const Demo = () => {
  const toast = useToast();
  return (
    <div className="flex flex-wrap gap-2 justify-center w-full">
      <Button
        type="primary"
        onClick={() => {
          toast({
            status: "primary",
            message: "Primary 토스트 알림입니다.",
            desc: "우측 하단에 메시지가 나타납니다.",
          });
        }}
      >
        Primary
      </Button>
    </div>
  );
};`}
/>
